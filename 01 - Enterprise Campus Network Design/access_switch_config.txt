! =====================================================================
! Cisco Catalyst C9300L-48P — Access Switch Stack (Layer 2)
! Sample / Sanitized Configuration for Portfolio Reference
! Note: IP addressing below uses RFC 5737 documentation ranges
!       (192.0.2.0/24, 198.51.100.0/24) — not real client addressing.
! =====================================================================

hostname ACCESS-STACK-11

!
! --- Global Security Hardening ---
!
service password-encryption
no ip http server
no ip http secure-server
enable secret 5 $1$XXXX$XXXXXXXXXXXXXXXXXXXXXX

!
! --- Switch Stack ---
!
switch 1 provision c9300l-48p-4x
switch 2 provision c9300l-48p-4x
switch 3 provision c9300l-48p-4x

!
! --- VLAN Definitions ---
!
vlan 10
 name DATA-VLAN
vlan 20
 name VOICE-VLAN
vlan 99
 name MGMT-VLAN

!
! --- Management Interface (Dedicated OOB) ---
!
interface Vlan99
 description Dedicated-Management-Interface
 ip address 198.51.100.251 255.255.255.0
 no shutdown
!
ip default-gateway 198.51.100.254

!
! --- Uplink to Core Switch (Port-Channel / LACP) ---
!
interface range TenGigabitEthernet1/1/1, TenGigabitEthernet2/1/1, TenGigabitEthernet3/1/1
 description Po11-Uplink-to-Core
 switchport mode trunk
 switchport trunk allowed vlan 10,20,99
 channel-protocol lacp
 channel-group 11 mode active
!
interface Port-channel11
 description Po11-Uplink-to-Core
 switchport mode trunk
 switchport trunk allowed vlan 10,20,99

!
! --- Access Ports: End-User / Data + Voice ---
!
interface range GigabitEthernet1/0/1-44
 description Access-Port-Data-Voice
 switchport mode access
 switchport access vlan 10
 switchport voice vlan 20
 spanning-tree portfast
 spanning-tree bpduguard enable
 storm-control broadcast level 20.00
 storm-control action trap
 switchport port-security
 switchport port-security maximum 3
 switchport port-security violation restrict
 switchport port-security mac-address sticky

!
! --- Unused Ports: Administratively Shut Down ---
!
interface range GigabitEthernet1/0/45-48
 description Unused-Port
 switchport mode access
 switchport access vlan 999
 shutdown

!
! --- Spanning Tree ---
!
spanning-tree mode rapid-pvst

!
! --- Secure Management Access ---
!
line vty 0 15
 transport input ssh
 login local
!
ip ssh version 2
!
username netadmin privilege 15 secret 5 $1$YYYY$YYYYYYYYYYYYYYYYYYYYYY

!
end
